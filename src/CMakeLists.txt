message("Getting source files in ${SOURCE_DIR}")

file(GLOB HEADERS "${SOURCE_DIR}/*.h")
file(GLOB SOURCES "${SOURCE_DIR}/*.cpp")

file(GLOB FRAMEWORK_HEADERS "${SOURCE_DIR}/framework/*.h")
file(GLOB FRAMEWORK_SOURCES "${SOURCE_DIR}/framework/*.cpp")

list(APPEND SOURCES ${SOURCES} ${FRAMEWORK_SOURCES})
list(APPEND HEADERS ${HEADERS} ${FRAMEWORK_HEADERS})

set(HEADERS ${HEADERS} PARENT_SCOPE)
set(SOURCES ${SOURCES} PARENT_SCOPE)
